　　4.

char   *   strcpy(char   *   pstrDest,const   char   *   pstrSource)
{
  assert((pstrDest!=NULL)&&(pstrSource!=NULL));  
  char   *   pstr=pstrDest;
  while((*(pstrDest++)=*(pstrSource++))!= '\0 ');
                return   pstr;
}

char   *   strrev(char   *   pstr)
{
  assert(pstr!=NULL);
 
  char   *   p=pstr;
  char   *   pret=pstr;
  while(*(p++)!= '\0 ');
  p--;
  char   tmp;
  while(p> pstr)
  {
    tmp=*p;
    *(p--)=*(pstr);
    *(pstr++)=tmp;    
  }
  return   pret;
